deseret_missionaries = {
	is_shown = {
		exists = c:DES
		THIS = c:DES
		c:DES = {
			total_population <= 100000
		}
	}

	possible = {
		NOT = {
			has_variable = recent_mission
			has_law = law_type:law_ethnostate
			total_population > 100000
		}
	}

	ai_chance = {
		base = 60
	}

	when_taken = {
		trigger_event = missionary.2
		set_variable = {
			name = recent_mission
			months = 18
		}
	}
}
