je_madagascar_autarky = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	immediate = {
		
	}
	
	complete = {
		has_healthy_economy = yes
		country_gdp_per_capita >= great_power_gdp_per_capita #Should be the average GDP per capita of the great powers
		market = {
			NOT = {
				any_market_goods = {
					OR = {
						is_consumed_by_government_buildings = yes
						is_consumed_by_military_buildings = yes
					}
					num_import_trade_routes > 0
				}
			}
			market_has_goods_shortage = no
		}
		OR = {
			has_law = law_type:law_isolationism
			market = {
				any_market_goods = {
					market_goods_exports >= 3500
				}
			}
		}
		is_subject = no
	}
	
	on_complete = {
		trigger_event = { id = madagascar_econ.2 }
	}

	fail = {
		is_subject = yes
	}

	on_fail = {
		trigger_event = { id = madagascar_econ.3 }
	}

	inheritable = yes

	weight = 400
}
