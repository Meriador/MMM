ig_devout2 = {
	color = { 229 228 226 }
	texture = "gfx/interface/icons/ig_icons/devout2.dds"
	
	ideologies = {
		ideology_charitable
		ideology_evangelist
		ideology_familialist
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_principled_society
		ig_trait_benevolent_empire
		ig_trait_moral_outrage
	}
	
	on_enable = {}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		religion = {
			shares_trait_with_state_religion = ROOT.owner
		}
	} #This does not actually work mechanically, that's handled below. It does make a tooltip show up that accurately describes my intention though, so it's useful for that.
	
	pop_weight = {
		value = 10
		
		if = {
			limit = {
				owner = {
					is_country_type = decentralized
				}
			}
			multiply = { value = 0 } 
		}

		elseif = {
			limit = { 
				NOT = {
					religion = {
						shares_heritage_trait_with_state_religion = ROOT.owner
					}
				}
			}
			multiply = { value = 0 } #this is why pops of the wrong religions don't join
		}
		
		else = {
			add = {
				desc = "POP_CLERGYMEN_URBAN"
				if = {
					limit = { 
						is_pop_type = clergymen
						pop_employment_building = building_urban_center
					}
					value = 300
				}
			}

			add = {
				desc = "POP_CLERGYMEN_LAWS"
				if = {
					limit = {
						is_pop_type = clergyman
					}
					if = {
						limit = {
							owner = {
								has_law = law_type:law_freedom_of_conscience
							}
						}
						value = 50
					}
					else_if = {
						limit = {
							owner = {
								has_law = law_type:law_total_separation
							}
						}
						value = 100
					}
				}
			}

			add = {
				desc = "POP_CLERGYMEN_RELIGION"
				if = {
					limit = {
						is_pop_type = clergyman
						is_state_religion = no
					}
					value = 200
				}
			}
			
			add = {
				desc = "POP_SHOPKEEPERS_RELIGION"
				if = {
					limit = { 
						is_pop_type = shopkeepers
						is_state_religion = no
					}
					value = 50
				}
			}

			add = {
				desc = "POP_CLERKS"
				if = {
					limit = { 
						is_pop_type = clerks
					}
					value = 50
				}
			}

			add = {
				desc = "POP_BUREAUCRATS"
				if = {
					limit = { 
						is_pop_type = bureaucrats
					}
					value = 25
				}
			}
			
			add = {
				desc = "POP_BUREAUCRATS_RELIGION"
				if = {
					limit = { 
						is_pop_type = bureaucrats
						is_state_religion = yes
					}
					value = 25
				}
			}
			
			add = {
				desc = "LITERACY_50"
				
				if = {
					limit = { 
						literacy_rate >= 0.50
					}
					add = 2
				}
			}
			
			add = {
				desc = "LITERACY_60"
				
				if = {
					limit = { 
						literacy_rate >= 0.60
					}
					add = 2
				}
			}
			
			add = {
				desc = "LITERACY_70"
				
				if = {
					limit = { 
						literacy_rate >= 0.70
					}
					add = 2
				}
			}
			
			add = {
				desc = "LITERACY_80"
				
				if = {
					limit = { 
						literacy_rate >= 0.80
					}
					add = 2
				}
			}
			
			add = {
				desc = "LITERACY_90"
				
				if = {
					limit = { 
						literacy_rate >= 0.90
					}
					add = 2
				}
			}
	
			multiply = {
				desc = "LEADER_POPULARITY"
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.0025
						add = 1
						max = 1.25
						min = 0.75
					}
				}
			}
		}
	}
	
	monarch_weight = {
		value = 0.25
	}
	
	commander_weight = {
		value = 1.0
	}
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
