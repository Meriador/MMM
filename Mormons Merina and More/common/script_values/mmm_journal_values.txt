country_gdp_per_capita = {
	value = gdp
	
	if = {
		limit = {
			total_population > 0
		}
		divide = total_population
	}
}

great_power_gdp_per_capita = {
	value = 0
	
	every_country = {
		limit = {
			country_rank = rank_value:great_power
		}
		add = gdp
	}

	every_country = {
		limit = {
			country_rank = rank_value:great_power
		}
		divide = total_population
	}
}
